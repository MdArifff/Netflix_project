-- 1. count hte number of movies and tv shows
select type,
     count(*) as total_content

-- 2. Find the most common rating for moives and TV shows
select type,
     rating,
	 count(*)
from netflix4
     group by 1,2
     order by 3 desc
     
select type,
     rating,
	 count(*),
	 rank() over(partition by type order by count(*)desc) as ranking  
from netflix4
     group by 1,2
     --order by 3 desc

-- sub query 
select type,
      rating
from 
(
 select type,
     rating,
	 count(*),
	 rank() over(partition by type order by count(*)desc) as ranking  
from netflix4
     group by 1,2
 )   as t1
where 
      ranking = 1



-- 3. List all movies released in a specific year (e.g, 2020) 
-- filter 2020
-- movies

select * from netflix4
where 
    type = 'Movie'
and
    release_year = 2020


-- Find the top 5 countries with the most conytent on netflix

select 
	country,
	count(show_id) as total_content
from netflix4
group by 1

-- this to separate the country names in one column (UNNEST)
SELECT
	string_to_array(country,',') as new_country
	from netflix4

-- UNNEST very important
SELECT
      unnest(string_to_array(country,',')) as new_country
	from netflix4

-- Final query
SELECT
      unnest(string_to_array(country,',')) as new_country,
	  count(show_id) as total_content
	from netflix4
group by 1
order by 2 desc
limit 5
